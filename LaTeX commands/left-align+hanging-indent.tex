% Note that this adds extra indentation to the generated reference section.
% To reset the reference section indentation, \setlength{\RaggedRightLeftskip}{0em} must be added to the end of the document.

\usepackage[document]{ragged2e}					% Sets whole document to ragged right (left-aligned)
	\setlength{\RaggedRightLeftskip}{1.5em}		% Sets left-alignment point, effectively indenting all lines
	\setlength{\RaggedRightParindent}{-1.5em}	% Offsets first line of paragraph back to normal alignment, effectively unindenting those lines
\usepackage{indentfirst}						% Applies indentation rules to first paragraph (LaTeX by default does not indent first paragraphs of sections)
